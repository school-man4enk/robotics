import time

led = Pin(2, Pin.OUT)

button = Pin(4, Pin.IN, Pin.PULL_UP)

while True:
    if not button.value():

        buzzer.duty(512)
        led.on()
        time.sleep(0.2)
        led.off()
        time.sleep(0.2)
    else:
        buzzer.deinit()
        led.off()
    time.sleep(0.1)
